1. Get the application URL by running these commands:
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ .Release.Name }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT

2. Access the service from within the cluster:
   DNS name: {{ .Release.Name }}.{{ .Release.Namespace }}.svc.cluster.local:30000

3. Access via Ingress:
   Host: tu-model-serving.com
   Paths:
     - http://tu-model-serving.com/docs
     - http://tu-model-serving.com/predict
     - http://tu-model-serving.com/openapi.json
